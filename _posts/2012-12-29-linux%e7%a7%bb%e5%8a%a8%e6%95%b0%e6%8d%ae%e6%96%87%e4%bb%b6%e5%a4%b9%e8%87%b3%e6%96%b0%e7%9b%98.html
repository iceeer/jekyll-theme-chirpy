-----挂载数据库盘
fdisk -l
fdisk /dev/xvdb
根据提示，依次输入“n”，“p”“1”，两次回车，“wq”，

mkfs.ext4 /dev/xvdb1

mkdir /var/www
echo '/dev/xvdb1 /data ext4 defaults 0 0' >> /etc/fstab
cat /etc/fstab
mount -a”命令挂载新分区，然后用“df -h”命令查看

----复制数据
cd /data
mkdir tmp
chmod 1777 tmp
mkdir var
mkdir home
cd /var
cp -ax * /data/var
cd /home
cp -ax * /data/home
cd /
cp var var.old
cp tmp tmp.old
cp home home.old
ln -s /data/var /var
ln -s /data/tmp /tmp
ln -s /data/home /home

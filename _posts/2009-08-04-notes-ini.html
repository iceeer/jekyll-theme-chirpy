http_pwd_change_cache_hours=1 ;强行将老密码有效期改为一个小时

HTTP_PWD_CHANGE_CACHE_HOURS=0 禁用HTTP密码缓存功能
如果要立即清理HTTP密码缓存，可以在控制台执行命令：show nlcache reset

Client_Clock=1 ;Enables the RPC's (from the client to a server) to be outputed to a file or the screen (0= Disabled)

ADMIN_CLIENT_SKIP_DOMINO=1 ;IBM Lotus Administrator打开文件项时不扫描domino文件夹内容

HTTPAllowDecodedUrlPercent=1 ; 解决Domino禁止URL地址包含百分号(%)

agentThreadDebug=1 ;值设为1或者0，1表示打开代理调试模式
AMgr_SchedulingInterval=10 ;设置代理管理器安排定时代理的最小时间间隔,10表示最小间隔为10分钟。AMgr_SchedulingInterval的有效值是1到60分钟
Amgr_WeekendDays=1,6,7 ;指定代理管理器的“周末”时间,在定时代理中可以设定该代理周末不运行,“周末”时间Amgr_WeekendDays的默认值是周六、周日
DominoAsynchronizeAgents = 1 ;平行的运行WEB触发的代理。例如，如果有一个WEBQUERYOPEN代理，同时运行这个代理的几率很大，但是NOTES不支持同时运行同一个代理，因此，如果三个用户同时申请了运行这个代理，那么只有一个用户的申请被批准，其余的只有等待。用了以上的设置之后，三个用户可以同时运行这个代理。

FormulaTimeout=60 ;设置公式运行时间限制,60表示公式最大执行时间为60秒，默认为无限制
Platform_Statistics_Enabled=1 ;让Domino最大限度地获得系统的内存，Domino 5.0.2（及更高版本）的这一属性支持Windows NT和Solaris平台
GOFASTER=1 ;设置加快视图索引速度,0表示放慢

NoWebFileSystemACLs=1 ;允许匿名用户访问dominoHTML目录

DominoCompleteDoctype改变Domino生成的HTML 标签(Domino 6.5.3+支持)，当出现HTML代码、游览器相同但显示效果不同时试着修改一下 标签看看。可取的值为：
0;
1;
2;
PS:也可以通过修改$$HTMLFrontMatter域（Domino 7.0.2+支持）

Server_Session_Timeout =30 ;30分钟后，如果用户 端（WEB）处于空闲状态，则自动关闭用户客户端到服务器端的连接，并释放资源。当用户重新操作客户端时，可再次创建新的连接（无需重输口令）

ServerTasks=;程序运行时运行的任务
ServerTasksAt0=;0点计划运行的任务（ServerTasksAt23则是23点时计划运行的任务）

TCPIP=TCP, 0, 15, 0,,12288,
;参数由6个部分组成，参数之间使用逗号（，）分隔。从左到右各个位置上的配置值分别是：

argv[0] — Domino支持的协议名称，如：TCP、Netbios等。
argv[1] — 协议绑定的网络适配编号（未使用）。
argv[2] — 连接会话的请求编号（未使用）。
argv[3] — 数据缓冲区大小。若配置为0，则数据缓冲区大小使用缺省值。对于不同的协议，分配的缺省数据缓冲区大小也不相同。
TCP：缺省大小为 8000
XPC：缺省大小为 2000
Netbios：缺省值为 8000
argv[4] — 指定预分配的网络缓冲区的数目（未使用）。
argv[5] — 端口功能启用标志。
未启用加密与压缩功能标志值为 0x3000 （12288）
加密功能标志值为 0x8000 （32768）
压缩功能标志值为 0x0020 （32）
argv[5]的值，由其中某两个标志位值相加而来的。启用加密与压缩功能时，其值为 0x8000 + 0x0020 = 0x8020 （32800）；仅启用压缩功能时，其值为：0x0020 + 0x3000 = 0x3020 （12320）；同理，仅启用加密功能时，其值为： 0x8000 + 0x3000 = 0xB000（45056）
